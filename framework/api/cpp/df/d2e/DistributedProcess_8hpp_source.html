261  return nullptr;

263  // ADD UID FOR ARCHIVING HISTORY
264  size_t uid;
265  if (invoke->has("_History_uid") == false)
266  {
267  // ISSUE UID AND ATTACH IT TO INVOKE'S LAST PARAMETER
268  uid = ((parallel::base::IParallelSystemArray*)system_array_)->_Fetch_history_sequence();
269  invoke->emplace_back(new InvokeParameter("_History_uid", uid));
270  }
271  else
272  {
273  // INVOKE MESSAGE ALREADY HAS ITS OWN UNIQUE ID
274  // - system_array_ IS A TYPE OF DistributedSystemArrayMediator. THE MESSAGE HAS COME FROM ITS MASTER
275  // - A Distributed HAS DISCONNECTED. THE SYSTEM SHIFTED ITS CHAIN TO ANOTHER SLAVE.
276  uid = invoke->get("_History_uid")->getValue<size_t>();
277 
278  // FOR CASE 1. UPDATE HISTORY_SEQUENCE TO MAXIMUM
279  if (uid > ((parallel::base::IParallelSystemArray*)system_array_)->_Get_history_sequence())
280  ((parallel::base::IParallelSystemArray*)system_array_)->_Set_history_sequence(uid);
281 
282  // FOR CASE 2. ERASE ORDINARY PROGRESSIVE HISTORY FROM THE DISCONNECTED
283  progress_list_.erase(uid);
284  }
285 
286  // ADD ROLE NAME FOR MEDIATOR
287  if (invoke->has("_Process_name") == false)
288  invoke->emplace_back(new InvokeParameter("_Process_name", name));
289 
290  // FIND THE MOST IDLE SYSTEM
291  shared_ptr<DistributedSystem> idle_system;
292 
293  for (size_t i = 0; i < system_array_->size(); i++)
294  {
295  shared_ptr<DistributedSystem> system = system_array_->at(i);
296  if (system->_Is_excluded() == true)
297  continue; // BEING REMOVED SYSTEM
298 
299  if (idle_system == nullptr
300  || system->_Get_progress_list()->size() < idle_system->_Get_progress_list()->size()
301  || system->getPerformance() < idle_system->getPerformance())
302  idle_system = system;
303  }
304

307

310 
311  // SEND DATA
