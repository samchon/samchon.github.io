241  static auto numberFormat(double val, int precision = 2) -> std::string
242  {
243  //IF VALUE IS ZERO OR NULL
244  if (val == 0.0)
245  return "0";
246  else if (val == INT_MIN)
247  return "";
248 
249  std::string str = "";
250 
251  //SETTING
252  bool isNegative = (val < 0);
253  val = abs(val);
254 
255  int cipher = (int)log10(val) + 1;
256 
257  //PRECISION
258  if (val != (unsigned long long)val)
259  {
260  int pointValue = (int)round((val - (unsigned long long)val) * pow(10.0, (double)precision));
261  str = "." + std::to_string(pointValue);
262  }
263 
264  //NATURAL NUMBER
265  for (int i = 0; i < cipher; i++)
266  {
267  int num = (int)((unsigned long long)val % (unsigned long long)pow(10.0, i + 1.0)); //TRUNCATE UPPER DIGIT VALUE
268  num = (int)(num / pow(10.0, (double)i));
269 
270  str = (char)(num + '0') + str;
271  if ((i + 1) % 3 == 0 && i < cipher - 1)
272  str = "," + str;
273  }
274 
275  //NEGATIVE NUMBER
276  if (isNegative == true)
277  str = "-" + str;
278 
279  return str;
280  };
