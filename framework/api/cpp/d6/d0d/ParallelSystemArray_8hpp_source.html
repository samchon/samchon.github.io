173  virtual auto sendPieceData(std::shared_ptr<protocol::Invoke> invoke, size_t first, size_t last) -> size_t
174  {
175  if (invoke->has("_History_uid") == false)
176  invoke->emplace_back(new protocol::InvokeParameter("_History_uid", _Fetch_history_sequence()));
177  else
178  {
179  // INVOKE MESSAGE ALREADY HAS ITS OWN UNIQUE ID
180  // - THIS IS A TYPE OF ParallelSystemArrayMediator. THE MESSAGE HAS COME FROM ITS MASTER
181  // - A ParallelSystem HAS DISCONNECTED. THE SYSTEM SHIFTED ITS CHAIN TO OTHER SLAVES.
182  size_t uid = invoke->get("_History_uid")->getValue<size_t>();
183 
184  // FOR CASE 1. UPDATE HISTORY_SEQUENCE TO MAXIMUM
185  if (uid > _Get_history_sequence())
186  _Set_history_sequence(uid);
187  }
188 
189  // TOTAL NUMBER OF PIECES TO DIVIDE
190  size_t segment_size = last - first;
191 
192  // SYSTEMS TO BE GET DIVIDED PROCESSES AND
193  std::vector<std::shared_ptr<ParallelSystem>> system_array;
194  std::vector<std::thread> threads; // THREADS TO EMBARK THEM
195 
196  system_array.reserve(size());
197  threads.reserve(size());
198 
199  // POP EXCLUDEDS
200  for (size_t i = 0; i < size(); i++)
201  if (at(i)->excluded_ == false)
202  system_array.push_back(at(i));
203 
204  // ORDERS
205  for (size_t i = 0; i < system_array.size(); i++)
206  {
207  std::shared_ptr<ParallelSystem> system = system_array.at(i);
208 
209  // COMPUTE FIRST AND LAST INDEX TO ALLOCATE
210  size_t piece_size = (i == system_array.size() - 1)
211  ? segment_size - first
212  : (size_t)(segment_size / system_array.size() * system->getPerformance());
213  if (piece_size == 0)
214  continue;
215 
216  // SEND DATA WITH PIECES' INDEXES
217  threads.emplace_back(&ParallelSystem::send_piece_data, system.get(), invoke, first, first + piece_size);
218  first += piece_size; // FOR THE NEXT STEP
219  }
220 
221  for (auto it = threads.begin(); it != threads.end(); it++)
222  it->join();
