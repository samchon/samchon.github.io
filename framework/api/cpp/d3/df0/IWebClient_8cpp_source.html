140 void IWebClient::listen()
141 {
142  shared_ptr<Invoke> binary_invoke;
143 
144  // CLIENT ADDS MASK
145  unsigned char mask = isServer() ? WebSocketUtil::MASK : 0;
146 
147  while (true)
148  {
149  try
150  {
151  size_t content_size = 0;
152 
154  // READ HEADER
156  array<unsigned char, 2> header_bytes;
157  unsigned char op_code; // false then binary
158  unsigned char size_header;
159 
160  socket->read_some(boost::asio::buffer(header_bytes));
161 
162  // INVALID PROTOCOL, UNMASKED MESSAGE
163  if (isServer() == true && header_bytes[1] < WebSocketUtil::MASK)
164  throw domain_error("unmasked message from client has delivered.");
165  else if (isServer() == false && header_bytes[1] >= WebSocketUtil::MASK)
166  throw domain_error("masked message from server has delivered.");
167 
168  // DECODE HEADER
169  op_code = header_bytes[0];
170  size_header = header_bytes.at(1) - mask;
171 
172  if (op_code == WebSocketUtil::DISCONNECT)
173  break; // DISCONNECTION SIGNAL HAS ARRIVED
174 
176  // READ CONTENT SIZE
178  if (size_header == (unsigned char)WebSocketUtil::TWO_BYTES)
179  {
180  // SIZE HEADER IS 2 BYTES
181  array<unsigned char, 2> size_bytes;
182  socket->read_some(boost::asio::buffer(size_bytes));
183 
184  for (size_t c = 0; c < size_bytes.size(); c++)
185  content_size += size_bytes[c] << (8 * (size_bytes.size() - 1 - c));
186  }
187  else if (size_header == (unsigned char)WebSocketUtil::EIGHT_BYTES)
188  {
189  // SIZE HEADER IS 8 BYTES
190  array<unsigned char, 8> size_bytes;
191  socket->read_some(boost::asio::buffer(size_bytes));
192 
193  for (size_t c = 0; c < size_bytes.size(); c++)
194  content_size += size_bytes[c] << (8 * (size_bytes.size() - 1 - c));
195  }
196  else
197  content_size = (size_t)size_header;
198 
200  // READ CONTENTS
202  if (op_code == WebSocketUtil::TEXT)
203  binary_invoke = listen_string(content_size);
204  else if (op_code == WebSocketUtil::BINARY)
205  listen_binary(content_size, binary_invoke);
206  else
207  throw domain_error("Unknown opcode has delivered.");
208  }
209  catch (exception &)
210  {
211  break;
212  }
213  }
