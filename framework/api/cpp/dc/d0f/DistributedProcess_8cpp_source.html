71 void DistributedProcess::sendData(shared_ptr<Invoke> invoke, double weight)
72 {
73  if (system_array_->empty() == true)
74  return;
75 
76  // ADD UID FOR ARCHIVING HISTORY
77  size_t uid;
78  if (invoke->has("_History_uid") == false)
79  {
80  // ISSUE UID AND ATTACH IT TO INVOKE'S LAST PARAMETER
81  uid = ++system_array_->history_sequence;
82  invoke->emplace_back(new InvokeParameter("_History_uid", uid));
83  }
84  else
85  {
86  // INVOKE MESSAGE ALREADY HAS ITS OWN UNIQUE ID
87  // - system_array_ IS A TYPE OF DistributedSystemArrayMediator. THE MESSAGE HAS COME FROM ITS MASTER
88  // - A Distributed HAS DISCONNECTED. THE SYSTEM SHIFTED ITS CHAIN TO ANOTHER SLAVE.
89  uid = invoke->get("_History_uid")->getValue<size_t>();
90 
91  // FOR CASE 1. UPDATE HISTORY_SEQUENCE TO MAXIMUM
92  if (uid > system_array_->history_sequence)
93  system_array_->history_sequence = uid;
94 
95  // FOR CASE 2. ERASE ORDINARY PROGRESSIVE HISTORY FROM THE DISCONNECTED
96  progress_list_.erase(uid);
97  }
98 
99  // ADD ROLE NAME FOR MEDIATOR
100  if (invoke->has("_Process_name") == false)
101  invoke->emplace_back(new InvokeParameter("_Process_name", name));
102 
103  // FIND THE MOST IDLE SYSTEM
104  shared_ptr<DistributedSystem> idle_system;
105 
106  for (size_t i = 0; i < system_array_->size(); i++)
107  {
108  shared_ptr<DistributedSystem> system = system_array_->at(i);
109  if (system->excluded_ == true)
110  continue; // BEING REMOVED SYSTEM
111 
112  if (idle_system == nullptr
113  || system->progress_list_.size() < idle_system->progress_list_.size()
114  || system->getPerformance() < idle_system->getPerformance())
115  idle_system = system;
116  }
117 
118  // ARCHIVE HISTORY ON PROGRESS_LIST (IN SYSTEM AND ROLE AT THE SAME TIME)
119  shared_ptr<DSInvokeHistory> history(new DSInvokeHistory(idle_system.get(), this, invoke, weight));
120 
121  progress_list_.emplace(uid, history);
122  idle_system->progress_list_.emplace(uid, make_pair(invoke, history));
123 
124  // SEND DATA
125  idle_system->sendData(invoke);
126 }
