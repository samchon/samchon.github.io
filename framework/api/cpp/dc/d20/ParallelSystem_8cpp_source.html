124 void ParallelSystem::_Send_back_history(shared_ptr<Invoke> invoke, shared_ptr<InvokeHistory> $history)
125 {
126  shared_ptr<PRInvokeHistory> history = dynamic_pointer_cast<PRInvokeHistory>($history);
127  if (history == nullptr)
128  return;
129 
130  // REMOVE UID AND FIRST, LAST INDEXES
131  for (size_t i = invoke->size(); i < invoke->size(); i--)
132  {
133  const string &name = invoke->at(i)->getName();
134 
135  if (name == "_History_uid" || name == "_Piece_first" || name == "_Piece_last")
136  invoke->erase(invoke->begin() + i);
137  }
138 
139  // RE-SEND (DISTRIBUTE) THE PIECE TO OTHER SLAVES
140  thread(&ParallelSystemArray::sendPieceData, getSystemArray(),
141  invoke, history->getFirst(), history->getLast()).detach();
142 
143  // ERASE FROM THE PROGRESS LIST
144  progress_list_.erase(history->getUID());
145 }
