81 {

95  //--------

97  vector<pair<shared_ptr<ParallelSystem>, double>> system_pairs;
98  double performance_index_average = 0.0;
99 
100  system_pairs.reserve(size());
101

103  {
104  shared_ptr<ParallelSystem> system = at(i);
105  if (system->history_list_.has(uid) == false)
106  continue; // NO HISTORY (HAVE NOT PARTICIPATED IN THE PARALLEL PROCESS)
107 
108  // COMPUTE PERFORMANCE INDEX BASIS ON EXECUTION TIME OF THIS PARALLEL PROCESS
109  shared_ptr<PRInvokeHistory> my_history = dynamic_pointer_cast<PRInvokeHistory>(system->history_list_.get(uid));
110  double performance_index = my_history->computeSize() / (double)my_history->computeElapsedTime();
111 
112  // PUSH TO SYSTEM PAIRS AND ADD TO AVERAGE
113  system_pairs.emplace_back(system, performance_index);
114  performance_index_average += performance_index;
115  }
116  performance_index_average /= system_pairs.size();
117 
118  // RE-CALCULATE PERFORMANCE INDEX
119  for (size_t i = 0; i < system_pairs.size(); i++)
120  {
121  // SYSTEM AND NEW PERFORMANCE INDEX BASIS ON THE EXECUTION TIME
122  shared_ptr<ParallelSystem> system = system_pairs[i].first;
123  if (system->enforced_ == true)
124  continue; // PERFORMANCE INDEX IS ENFORCED. DOES NOT PERMIT REVALUATION
125 
126  double new_performance = system_pairs[i].second / performance_index_average;
127 
128  // DEDUCT RATIO TO REFLECT THE NEW PERFORMANCE INDEX
129  double ordinary_ratio;
130  if (system->history_list_.size() < 2)
131  ordinary_ratio = .3;
132  else
133  ordinary_ratio = min(0.7, 1.0 / (system->history_list_.size() - 1.0));
134 
135  system->performance_ = (system->getPerformance() * ordinary_ratio) + (new_performance * (1 - ordinary_ratio));
136  }
137 
138  // AT LAST, NORMALIZE PERFORMANCE INDEXES OF ALL SLAVE SYSTEMS
139  _Normalize_performance();
