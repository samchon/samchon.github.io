81 void DistributedSystemArray::estimate_role_performance(shared_ptr<DSInvokeHistory> history)
82 {
83  DistributedProcess *process = history->getProcess();
84  if (process->enforced_ == true)
85  return; // THE RESOURCE INDEX IS ENFORCED. DO NOT PERMIT REVALUATION
86 
87  double average_elapsed_time_of_others = 0;
88  size_t denominator = 0;
89 
90  // COMPUTE AVERAGE ELAPSED TIME
91  for (auto it = process_map_.begin(); it != process_map_.end(); it++)
92  {
93  DistributedProcess *my_process = it->second.get();
94  if (my_process == process || my_process->history_list_.empty() == true)
95  continue;
96 
97  average_elapsed_time_of_others += my_process->compute_average_elapsed_time() * my_process->getResource();
98  denominator++;
99  }
100 
101  // COMPARE WITH THIS HISTORY'S ELAPSED TIME
102  if (denominator != 0)
103  {
104  // DIVE WITH DENOMINATOR
105  average_elapsed_time_of_others /= (double)denominator;
106 
107  // DEDUCT NEW PERFORMANCE INDEX BASED ON THE EXECUTION TIME
108  // - ROLE'S PERFORMANCE MEANS; HOW MUCH TIME THE ROLE NEEDS
109  // - ELAPSED TIME IS LONGER, THEN PERFORMANCE IS HIGHER
110  double new_performance = history->computeElapsedTime() / average_elapsed_time_of_others;
111 
112  // DEDUCT RATIO TO REFLECT THE NEW PERFORMANCE INDEX -> MAXIMUM: 15%
113  double ordinary_ratio;
114  if (process->history_list_.size() < 2)
115  ordinary_ratio = .15;
116  else
117  ordinary_ratio = min(.85, 1.0 / (process->history_list_.size() - 1.0));
118 
119  // DEFINE NEW PERFORMANCE
120  process->setResource
121  (
122  (process->getResource() * ordinary_ratio)
123  + (new_performance * (1 - ordinary_ratio))
124  );
125  }

134  size_t denominator = 0;
135 
136  // COMPUTE AVERAGE ELAPSED TIME
137  for (size_t i = 0; i < this->size(); i++)
138  {
139  shared_ptr<DistributedSystem> system = this->at(i);
140 
141  double avg = system->compute_average_elapsed_time();
142  if (avg == -1)
143  continue;
144 
145  average_elapsed_time_of_others += avg;
146  denominator++;
147  }
148 
149  // COMPARE WITH THIS HISTORY'S ELAPSED TIME
150  if (denominator != 0)

157  // - ELAPSED TIME IS LOWER, THEN PERFORMANCE IS HIGHER

176 void DistributedSystemArray::_Normalize_performance()
177 {

179  super::_Normalize_performance();
180 
181  // NORMALIZE ROLES' PERFORMANCE INDEXES
182  double average = 0;
183  size_t denominator = 0;

186  {
187  auto process = it->second;
188  if (process->enforced_ == true)
189  continue; // THE RESOURCE INDEX IS ENFORCED. DO NOT PERMIT REVALUATION

192  denominator++;
193  }

196  // DIVIDE FROM THE AVERAGE

198  {
199  auto process = it->second;
200  if (process->enforced_ == true)
201  continue; // THE RESOURCE INDEX IS ENFORCED. DO NOT PERMIT REVALUATION
202 
203  process->setResource(process->getResource() / average);
